  i, j, r;
  c : ARRAY[100];
  
{
   i = 1;
   WHILE i < 11
   {
       c[i] = i;
       i = i + 1;
   }
   i = 1;
   WHILE i < 11
   {
       j = 10;
       c[i + j] = i * i;
       i = i + 1;
   }
   i = 1;
   WHILE i < 21
   {
      j = i + 1;
      WHILE j < 21
      {
         IF c[i] > c[j]
         {
             r = c[i];
             c[i] = c[j];
             c[j] = r;
         }
         j = j + 1;
      }
      i = i + 1;
   }
   i = 1;
   WHILE i < 21
   {
       r = c[i];
       print r;
       i = i + 1;
   }
}
